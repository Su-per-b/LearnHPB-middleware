00027 /* ------------------------------------------------------------------------- 
00028  * main.c
00029  * Implements simulation of a single FMU instance using the forward Euler
00030  * method for numerical integration.
00031  * Command syntax: see printHelp()
00032  * Simulates the given FMU from t = 0 .. tEnd with fixed step size h and 
00033  * writes the computed solution to file 'result.csv'.
00034  * The CSV file (comma-separated values) may e.g. be plotted using 
00035  * OpenOffice Calc or Microsoft Excel. 
00036  * This progamm demonstrates basic use of an FMU.
00037  * Real applications may use advanced numerical solvers instead, means to 
00038  * exactly locate state events in time, graphical plotting utilities, support 
00039  * for coexecution of many FMUs, stepping and debug support, user control
00040  * of parameter and start values etc. 
00041  * All this is missing here.
00042  *
00043  * Revision history

00049  *  - 7z.exe 4.57 zip and unzip tool, see http://www.7-zip.org

00051  * -------------------------------------------------------------------------

00265       char* end = strchr(msg+i+1, '#');

00273         // ## detected, output #

00299   } // while
00300   buffer[k] = '\0';
